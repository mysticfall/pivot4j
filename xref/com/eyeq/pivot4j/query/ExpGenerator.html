21  /**
22   * Generate MDX expression for TreeNode
23   */
24  public class ExpGenerator {
25  
26  	private QuaxUtil quaxUtil;
27  
28  	private ExpNode rootNode;
29  
30  	private int nDimension;

32  	/**
33  	 * @param rootNode
34  	 * @param nDimension
35  	 * @param quaxUtil
36  	 */
37  	public void init(ExpNode rootNode, int nDimension, QuaxUtil quaxUtil) {
38  		this.rootNode = rootNode;
39  		this.nDimension = nDimension;
40  		this.quaxUtil = quaxUtil;
41  	}

43  	/**
44  	 * Generate MDX Expression
45  	 * 
46  	 * @return
47  	 */
48  	public Exp generate() {
49  		Exp exp = null;
50  
51  		List<TreeNode<Exp>> nodes = rootNode.getChildren();
52  
53  		// single members (nDimension = 1) are enclosed in set brackets
54  		// collect single members
55  		List<Exp> openSet = new ArrayList<Exp>();
56  
57  		// loop over top level nodes
58  		for (TreeNode<Exp> node : nodes) {
59  			Exp expForNode = genExpForNode(node, nDimension);
60  
61  			boolean closeOpenSet = false;
62  			if (nDimension == 1) {
63  				if (quaxUtil.isMember(expForNode)) {
64  					openSet.add(expForNode);
65  					continue;
66  				} else {
67  					closeOpenSet = true;
68  				}
69  			} else {
70  				if (quaxUtil.isFunCallTo(expForNode, "()")) {
71  					openSet.add(expForNode);
72  					continue;
73  				} else {
74  					closeOpenSet = true;
75  				}
76  			}
77  
78  			if (closeOpenSet && !openSet.isEmpty()) {
79  				// close open set
80  				Exp set = new FunCall("{}", Syntax.Braces, openSet);
81  
82  				if (exp == null) {
83  					exp = set;
84  				} else {
85  					// generate Union
86  					FunCall call = new FunCall("Union", Syntax.Function);
87  					call.getArgs().add(exp);
88  					call.getArgs().add(set);
89  
90  					exp = call;
91  				}
92  
93  				openSet.clear();
94  			}
95  
96  			if (exp == null) {
97  				exp = expForNode;
98  			} else {
99  				// generate Union of Exp and expForNode
100 				FunCall call = new FunCall("Union", Syntax.Function);
101 				call.getArgs().add(exp);
102 				call.getArgs().add(expForNode);
103 
104 				exp = call;
105 			}
106 		}
107 
108 		if (!openSet.isEmpty()) {
109 			// close open set
110 			Exp set = new FunCall("{}", Syntax.Braces, openSet);
111 
112 			if (exp == null) {
113 				exp = set;
114 			} else {
115 				// generate Union
116 				FunCall call = new FunCall("Union", Syntax.Function);
117 				call.getArgs().add(exp);
118 				call.getArgs().add(set);
119 
120 				exp = call;
121 			}
122 
123 			openSet.clear();
124 		}
125 
126 		return exp;
127 	}

129 	/**
130 	 * Recursively generate Exp for a node
131 	 * 
132 	 * @param node
133 	 * @return
134 	 */
135 	private Exp genExpForNode(TreeNode<Exp> node, int untilIndex) {
136 		Exp eNode = node.getReference();
137 		if (node.getLevel() == untilIndex) {
138 			// last dimension
139 			return eNode;
140 		}
141 
142 		// use tuple representation if possible
143 		Exp[] tuple = genTuple(node);
144 		if (tuple != null) {
145 			if (tuple.length == 1) {
146 				return tuple[0];
147 			} else {
148 				return new FunCall("()", Syntax.Parentheses,
149 						Arrays.asList(tuple));
150 			}
151 		}
152 
153 		// generate CrossJoin
154 		Exp exp = null;
155 
156 		List<TreeNode<Exp>> childNodes = node.getChildren();
157 		for (TreeNode<Exp> childNode : childNodes) {
158 			Exp childExp = genExpForNode(childNode, untilIndex);
159 
160 			Exp eSet;
161 			if (!quaxUtil.isMember(eNode)) {
162 				// FunCall
163 				eSet = eNode;
164 			} else {
165 				// member
166 				FunCall call = new FunCall("{}", Syntax.Braces);
167 				call.getArgs().add(eNode);
168 
169 				eSet = call;
170 			}
171 
172 			if (childExp == null) {
173 				exp = eSet;
174 			} else {
175 				Exp childSet = bracesAround(childExp);
176 
177 				FunCall cj = new FunCall("CrossJoin", Syntax.Function);
178 				cj.getArgs().add(eSet);
179 				cj.getArgs().add(childSet);
180 
181 				if (exp == null) {
182 					exp = cj;
183 				} else {
184 					FunCall call = new FunCall("Union", Syntax.Function);
185 					call.getArgs().add(exp);
186 					call.getArgs().add(cj);
187 
188 					exp = call;
189 				}
190 			}
191 		}
192 		return exp;
193 	}

195 	/**
196 	 * put braces around single member or single tuple
197 	 * 
198 	 * @param oExp
199 	 * @return set exp
200 	 */
201 	private Exp bracesAround(Exp oExp) {
202 		Exp oSet;
203 
204 		if (quaxUtil.isMember(oExp) || quaxUtil.isFunCallTo(oExp, "()")) {
205 			FunCall call = new FunCall("{}", Syntax.Braces);
206 			call.getArgs().add(oExp);
207 
208 			oSet = call;
209 		} else {
210 			oSet = oExp;
211 		}
212 
213 		return oSet;
214 	}

216 	/**
217 	 * generate Tuple Exp
218 	 * 
219 	 * @param node
220 	 * @return
221 	 */
222 	private Exp[] genTuple(TreeNode<Exp> node) {
223 		if (!quaxUtil.isMember(node.getReference())) {
224 			return null;
225 		}
226 
227 		int size = nDimension - node.getLevel() + 1;
228 		if (size == 1) {
229 			// single member
230 			return new Exp[] { node.getReference() };
231 		}
232 
233 		List<TreeNode<Exp>> childNodes = node.getChildren();
234 		if (childNodes.size() != 1) {
235 			return null;
236 		}
237 
238 		Exp[] nextTuple = genTuple(childNodes.get(0));
239 		if (nextTuple == null) {
240 			return null;
241 		}
242 
243 		Exp[] tupleMembers = new Exp[size];
244 		tupleMembers[0] = node.getReference();
245 		for (int i = 1; i < tupleMembers.length; i++) {
246 			tupleMembers[i] = nextTuple[i - 1];
247 		}
248 
249 		return tupleMembers;
250 	}
