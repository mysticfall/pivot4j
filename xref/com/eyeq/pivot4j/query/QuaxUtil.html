132 	 * Check, whether funcall set contains member
133 	 * 
134 	 * @param oExp
135 	 * @param member
136 	 * @return true if FunCall contains member
137 	 */
138 	public boolean isMemberInFunCall(Exp oExp, Member member)
139 			throws UnknownExpressionException {
140 		if (!isFunCall(oExp)) {
141 			return false;
142 		}
143 
144 		FunCall f = (FunCall) oExp;
145 
146 		if (f.isCallTo("Children")) {
147 			return isMemberInChildren(f, member);
148 		} else if (f.isCallTo("Descendants")) {
149 			return isMemberInDescendants(f, member);
150 		} else if (f.isCallTo("Members")) {
151 			return isMemberInLevel(f, member);
152 		} else if (f.isCallTo("Union")) {
153 			return isMemberInUnion(f, member);
154 		} else if (f.isCallTo("{}")) {
155 			return isMemberInSet(f, member);
156 		}
157 
158 		throw new UnknownExpressionException(f.getFunction());
159 	}

161 	/**
162 	 * Check, whether a funcall set contains any child of a specific member
163 	 * 
164 	 * @param oExp
165 	 * @param member
166 	 * @return true, if FunCall contains member's child
167 	 * @throws UnknownExpressionException
168 	 */
169 	public boolean isChildOfMemberInFunCall(Exp oExp, Member member)
170 			throws UnknownExpressionException {
171 		// calculated members do not have children
172 		if (!isFunCall(oExp) || member.isCalculated()) {
173 			return false;
174 		}
175 
176 		FunCall f = (FunCall) oExp;
177 
178 		if (f.isCallTo("Children")) {
179 			return OlapUtils.equals(member, memberForExp(f.getArgs().get(0)));
180 		} else if (f.isCallTo("Descendants")) {
181 			// true, if f = descendants(m2, level) contains any child of m
182 			// so level must be parent-level of m
183 			Member ancestor = memberForExp(f.getArgs().get(0));
184 
185 			Level level = levelForExp(f.getArgs().get(1));
186 			Level parentLevel = getParentLevel(level);
187 
188 			if (parentLevel != null
189 					&& OlapUtils.equals(member.getLevel(), parentLevel)) {
190 				int ancestorLevelNumber = ancestor.getLevel().getDepth();
191 				while (ancestorLevelNumber < member.getLevel().getDepth()) {
192 					member = member.getParentMember();
193 				}
194 
195 				if (OlapUtils.equals(member, ancestor)) {
196 					return true;
197 				} else {
198 					return false;
199 				}
200 			} else {
201 				return false;
202 			}
203 		} else if (f.isCallTo("Members")) {
204 			Level level = levelForExp(f.getArgs().get(0));
205 			Level parentLevel = null;
206 
207 			if (level.getDepth() > 0) {
208 				List<Level> levels = level.getHierarchy().getLevels();
209 				for (Level l : levels) {
210 					if (l.getDepth() == level.getDepth() - 1) {
211 						parentLevel = l;
212 						break;
213 					}
214 				}
215 			}
216 
217 			if (parentLevel != null
218 					&& OlapUtils.equals(member.getLevel(), parentLevel)) {
219 				return true;
220 			} else {
221 				return false;
222 			}
223 		} else if (f.isCallTo("Union")) {
224 			if (isChildOfMemberInFunCall(f.getArgs().get(0), member)) {
225 				return true;
226 			} else {
227 				return isChildOfMemberInFunCall(f.getArgs().get(1), member);
228 			}
229 		} else if (f.isCallTo("{}")) {
230 			for (Exp exp : f.getArgs()) {
231 				Member mm = memberForExp(exp);
232 
233 				Member mmp = mm.getParentMember();
234 				if (mmp != null && OlapUtils.equals(mmp, member)) {
235 					return true;
236 				}
237 			}
238 			return false;
239 		}
240 
241 		throw new UnknownExpressionException(f.getFunction());
242 	}

244 	/**
245 	 * Check, whether funcall set contains descendant of a specific member
246 	 * 
247 	 * @param oExp
248 	 * @param member
249 	 * @return true if FunCall contains descendant of member
250 	 */
251 	public boolean isDescendantOfMemberInFunCall(Exp oExp, Member member)
252 			throws UnknownExpressionException {
253 		// calculated members do not have children
254 		if (!isFunCall(oExp) || member.isCalculated()) {
255 			return false;
256 		}
257 
258 		FunCall f = (FunCall) oExp;
259 
260 		if (f.isCallTo("Children")) {
261 			// true, if m2.children contains descendants of m
262 			// <==> m is equal or ancestor of m2
263 			Member mExp = memberForExp(f.getArgs().get(0));
264 			return OlapUtils.equals(member, mExp) || isDescendant(member, mExp);
265 		} else if (f.isCallTo("Descendants")) {
266 			// true, if descendants(m2) contain descendants of m
267 			// <==> m is equal or ancestor of m2
268 			Member mExp = memberForExp(f.getArgs().get(0));
269 			return OlapUtils.equals(member, mExp) || isDescendant(member, mExp);
270 		} else if (f.isCallTo("Members")) {
271 			Level levExp = levelForExp(f.getArgs().get(0));
272 			return levExp.getDepth() > member.getLevel().getDepth();
273 		} else if (f.isCallTo("Union")) {
274 			if (isDescendantOfMemberInFunCall(f.getArgs().get(0), member)) {
275 				return true;
276 			} else {
277 				return isDescendantOfMemberInFunCall(f.getArgs().get(1), member);
278 			}
279 		} else if (f.isCallTo("{}")) {
280 			for (Exp arg : f.getArgs()) {
281 				Member mExp = memberForExp(arg);
282 				return (!member.equals(mExp) && isDescendant(member, mExp));
283 			}
284 
285 			return false;
286 		}
287 
288 		throw new UnknownExpressionException(f.getFunction());
289 	}

304 	/**
305 	 * @param ancestor
306 	 * @param descendant
307 	 */
308 	public boolean isDescendant(Member ancestor, Member descendant) {
309 		// a calculated member, even if defined under "ancestor" is *not*
310 		// descendant,
311 		// WITM MEMBER a.b as '..'
312 		// a.children does *not* include b
313 		if (descendant.isCalculated()) {
314 			return false;
315 		}
316 		if (OlapUtils.equals(ancestor, descendant)) {
317 			return false;
318 		}
319 
320 		int ancestorLevelNumber = ancestor.getDepth();
321 		Member mm = descendant;
322 		while (mm != null && ancestorLevelNumber < mm.getDepth()) {
323 			mm = mm.getParentMember();
324 		}
325 
326 		if (OlapUtils.equals(mm, ancestor)) {
327 			return true;
328 		} else {
329 			return false;
330 		}
331 	}

333 	/**
334 	 * Check whether a Funcall does NOT resolve to top level of hierarchy
335 	 * 
336 	 * @param oExp
337 	 *            - FunCall Exp
338 	 * @return true, if any member of the set defined by funcall is NOT top
339 	 *         level
340 	 */
341 	public boolean isFunCallNotTopLevel(Exp oExp)
342 			throws UnknownExpressionException {
343 		if (!isFunCall(oExp)) {
344 			return false;
345 		}
346 
347 		FunCall f = (FunCall) oExp;
348 
349 		if (f.isCallTo("Children")) {
350 			return true; // children *not* top level
351 		} else if (f.isCallTo("Descendants")) {
352 			return true; // descendants*not* top level
353 		} else if (f.isCallTo("Members")) {
354 			Level level = levelForExp(f.getArgs().get(0));
355 			return (level.getDepth() > 0);
356 		} else if (f.isCallTo("Union")) {
357 			if (isFunCallNotTopLevel(f.getArgs().get(0))) {
358 				return true;
359 			}
360 			return isFunCallNotTopLevel(f.getArgs().get(1));
361 		} else if (f.isCallTo("{}")) {
362 			for (Exp exp : f.getArgs()) {
363 				if (!isMemberOnToplevel(exp)) {
364 					return true;
365 				}
366 			}
367 			return false;
368 		}
369 
370 		throw new UnknownExpressionException(f.getFunction());
371 	}

389 	/**
390 	 * Check a Funcall expression whether we can handle it. currently we can
391 	 * basically handle following FunCalls member.children, member.descendants,
392 	 * level.members
393 	 */
394 	public boolean canHandle(Exp oExp) {
395 		if (isMember(oExp)) {
396 			return true;
397 		} else if (isFunCall(oExp)) {
398 			FunCall f = (FunCall) oExp;
399 
400 			if (f.isCallTo("children")) {
401 				return true;
402 			}
403 			if (f.isCallTo("descendants")) {
404 				return true;
405 			}
406 			if (f.isCallTo("members")) {
407 				return true;
408 			}
409 			if (f.isCallTo("{}")) {
410 				return true;
411 			}
412 			if (f.isCallTo("union")) {
413 				for (Exp exp : f.getArgs()) {
414 					if (!canHandle(exp)) {
415 						return false;
416 					}
417 				}
418 				return true;
419 			}
420 		}
421 
422 		return false;
423 	}

437 	public List<Exp> getChildMembers(Exp oParent) {
438 		Member parent = memberForExp(oParent);
439 
440 		if (parent == null) {
441 			return Collections.emptyList();
442 		}
443 
444 		List<? extends Member> members;
445 
446 		try {
447 			members = parent.getChildMembers();
448 		} catch (OlapException e) {
449 			throw new PivotException(e);
450 		}
451 
452 		List<Exp> children = new ArrayList<Exp>(members.size());
453 		for (Member member : members) {
454 			children.add(expForMember(member));
455 		}
456 
457 		return children;
458 	}

489 	/**
490 	 * @param oExp
491 	 * @return
492 	 */
493 	public StringBuilder funString(Exp oExp) {
494 		FunCall f = (FunCall) oExp;
495 
496 		StringBuilder sb = new StringBuilder();
497 
498 		if (f.isCallTo("Children")) {
499 			Member m = memberForExp(f.getArgs().get(0));
500 			sb.append(m.getUniqueName());
501 			sb.append(".Children");
502 		} else if (f.isCallTo("Descendants")) {
503 			Member m = memberForExp(f.getArgs().get(0));
504 			Level lev = levelForExp(f.getArgs().get(1));
505 			sb.append("Descendants(");
506 			sb.append(m.getUniqueName());
507 			sb.append(",");
508 			sb.append(lev.getUniqueName());
509 			sb.append(")");
510 		} else if (f.isCallTo("members")) {
511 			Level lev = levelForExp(f.getArgs().get(0));
512 			sb.append(lev.getUniqueName());
513 			sb.append(".Members");
514 		} else if (f.isCallTo("Union")) {
515 			sb.append("Union(");
516 			FunCall f1 = (FunCall) f.getArgs().get(0);
517 			sb.append(funString(f1));
518 			sb.append(",");
519 			FunCall f2 = (FunCall) f.getArgs().get(1);
520 			sb.append(funString(f2));
521 			sb.append(")");
522 		} else if (f.isCallTo("{}")) {
523 			sb.append("{");
524 
525 			boolean isFollow = false;
526 			for (Exp exp : f.getArgs()) {
527 				if (isFollow) {
528 					sb.append(",");
529 				} else {
530 					isFollow = true;
531 				}
532 
533 				Member m = memberForExp(exp);
534 				sb.append(m.getUniqueName());
535 			}
536 			sb.append("}");
537 		} else if (f.isCallTo("TopCount") || f.isCallTo("BottomCount")
538 				|| f.isCallTo("TopPercent") || f.isCallTo("BottomPercent")) {
539 			// just generate Topcount(set)
540 			sb.append(f.getFunction());
541 			sb.append("(");
542 			FunCall f1 = (FunCall) f.getArgs().get(0);
543 			sb.append(funString(f1));
544 			sb.append(")");
545 		}
546 		return sb;
547 	}

647 	/**
648 	 * @param oExp
649 	 * @return hierarchy for Exp
650 	 * @throws UnknownExpressionException
651 	 */
652 	public Hierarchy hierForExp(Exp oExp) throws UnknownExpressionException {
653 		if (isMember(oExp)) {
654 			return memberForExp(oExp).getHierarchy();
655 		} else if (oExp instanceof SetExp) {
656 			// set expression generated by CalcSet extension
657 			SetExp set = (SetExp) oExp;
658 			return set.getHierarchy();
659 		}
660 
661 		// must be FunCall
662 		FunCall f = (FunCall) oExp;
663 
664 		if (f.isCallTo("Children") || f.isCallTo("Descendants")
665 				|| f.isCallTo("{}")) {
666 			Member member = memberForExp(f.getArgs().get(0));
667 			return member.getHierarchy();
668 		} else if (f.isCallTo("Members")) {
669 			Level level = levelForExp(f.getArgs().get(0));
670 			return level.getHierarchy();
671 		} else if (f.isCallTo("Union")) {
672 			// continue with first set
673 			return hierForExp(f.getArgs().get(0));
674 		} else if (f.isCallTo("TopCount") || f.isCallTo("BottomCount")
675 				|| f.isCallTo("TopPercent") || f.isCallTo("BottomPercent")
676 				|| f.isCallTo("Filter")) {
677 			// continue with base set of top bottom function
678 			return hierForExp(f.getArgs().get(0));
679 		}
680 
681 		throw new UnknownExpressionException(f.getFunction());
682 	}

684 	/**
685 	 * @param hierarchy
686 	 *            the Hierarchy
687 	 * @param expandAllMember
688 	 *            if true, an "All" member will be expanded
689 	 * @return a set for the top level members of an hierarchy
690 	 */
691 	public Exp topLevelMembers(Hierarchy hierarchy, boolean expandAllMember) {
692 		Level topLevel = hierarchy.getLevels().get(0);
693 
694 		Member mAll;
695 		try {
696 			mAll = hierarchy.getDefaultMember();
697 
698 			if (!mAll.isAll()) {
699 				mAll = null;
700 				for (Member m : hierarchy.getRootMembers()) {
701 					if (m.isAll()) {
702 						mAll = m;
703 						break;
704 					}
705 				}
706 			}
707 		} catch (OlapException e) {
708 			throw new PivotException(e);
709 		}
710 
711 		// if there is an All Member, we will have to expand it
712 		// according to expandAllMember flag
713 		if (mAll != null) {
714 			List<Exp> memar = new ArrayList<Exp>();
