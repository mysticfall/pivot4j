76  	 * @param model
77  	 * @param parser
78  	 */
79  	public QueryAdapter(PivotModel model, MdxParser parser) {
80  		if (model == null) {
81  			throw new IllegalArgumentException(
82  					"Missing required argument 'model'.");
83  		}
84  
85  		if (parser == null) {
86  			throw new IllegalArgumentException(
87  					"Missing required argument 'parser'.");
88  		}
89  
90  		this.model = model;
91  		this.parsedQuery = parseQuery(model.getMdx());
92  
93  		QueryAxis[] queryAxes = parsedQuery.getAxes();
94  
95  		this.quaxes = new ArrayList<Quax>(queryAxes.length);
96  
97  		int ordinal = 0;
98  		for (@SuppressWarnings("unused")
99  		QueryAxis queryAxis : queryAxes) {
100 			Quax quax = new Quax(ordinal++, parser);
101 			quax.addChangeListener(quaxListener);
102 
103 			quaxes.add(quax);
104 		}
105 	}

233 	/**
234 	 * Update the Query Object before Execute. The current query is build from -
235 	 * the original query - adding the drilldown groups - apply pending swap
236 	 * axes - apply pending sorts.
237 	 */
238 	public ParsedQuery updateQuery() {
239 		// if quax is to be used, generate axes from quax
240 		if (useQuax) {
241 			int iQuaxToSort = activeQuaxToSort();
242 
243 			QueryAxis[] qAxes = parsedQuery.getAxes();
244 
245 			int i = 0;
246 			for (Quax quax : quaxes) {
247 				boolean doHierarchize = false;
248 				if (quax.isHierarchizeNeeded() && i != iQuaxToSort) {
249 					doHierarchize = true;
250 
251 					if (logger.isDebugEnabled()) {
252 						logger.debug("MDX Generation added Hierarchize()");
253 					}
254 				}
255 
256 				Exp eSet = (Exp) quax.genExp(doHierarchize);
257 				qAxes[i].setExp(eSet);
258 
259 				i++;
260 			}
261 		}
262 
263 		// generate order function if neccessary
264 		if (!useQuax) {
265 			// if Quax is used, the axis exp's are re-generated every time.
266 			// if not -
267 			// adding a sort to the query must not be permanent.
268 			// Therefore, we clone the orig state of the query object and
269 			// use
270 			// the clone furthermore in order to avoid duplicate "Order"
271 			// functions.
272 			if (cloneQuery == null) {
273 				if (isSortOnQuery()) {
274 					this.cloneQuery = (ParsedQuery) parsedQuery.clone();
275 				}
276 			} else {
277 				// reset to original state
278 				if (isSortOnQuery()) {
279 					this.parsedQuery = (ParsedQuery) cloneQuery.clone();
280 				} else {
281 					this.parsedQuery = cloneQuery;
282 				}
283 			}
284 
285 			addSortToQuery();
286 		}
287 
288 		// swap axes function if neccessary
289 		if (axesSwapped) {
290 			swapAxes();
291 		}
292 
293 		return parsedQuery;
294 	}

296 	/**
297 	 * Apply sort to query
298 	 */
299 	public void addSortToQuery() {
300 		if (isSortOnQuery()) {
301 			switch (model.getSortMode()) {
302 			case ASC:
303 			case DESC:
304 			case BASC:
305 			case BDESC:
306 				// call sort
307 				orderAxis(parsedQuery);
308 				break;
309 			case TOPCOUNT:
310 				topBottomAxis(parsedQuery, "TopCount");
311 				break;
312 			case BOTTOMCOUNT:
313 				topBottomAxis(parsedQuery, "BottomCount");
314 				break;
315 			default:
316 				return; // do nothing
317 			}
318 		}
319 	}

321 	/**
322 	 * Add Order Funcall to QueryAxis
323 	 * 
324 	 * @param monAx
325 	 * @param monSortMode
326 	 */
327 	protected void orderAxis(ParsedQuery pq) {
328 		// Order(TopCount) is allowed, Order(Order) is not permitted
329 		QueryAxis[] queryAxes = pq.getAxes();
330 		QueryAxis qa = queryAxes[quaxToSort.getOrdinal()];
331 
332 		Exp setForAx = qa.getExp();
333 
334 		// setForAx is the top level Exp of the axis
335 		// put an Order FunCall around
336 		Exp[] args = new Exp[3];
337 		args[0] = setForAx; // the set to be sorted is the set representing the
338 							// query axis
339 		// if we got more than 1 position member, generate a tuple for the 2.arg
340 		Exp sortExp;
341 
342 		List<Member> sortPosMembers = model.getSortPosMembers();
343 		if (sortPosMembers == null) {
344 			return;
345 		}
346 
347 		if (sortPosMembers.size() > 1) {
348 			Exp[] memberExp = new Exp[sortPosMembers.size()];
349 			for (int i = 0; i < memberExp.length; i++) {
350 				memberExp[i] = QuaxUtil.expForMember(sortPosMembers.get(i));
351 			}
352 
353 			sortExp = new FunCall("()", memberExp, Syntax.Parentheses);
354 		} else {
355 			sortExp = QuaxUtil.expForMember(sortPosMembers.get(0));
356 		}
357 
358 		args[1] = sortExp;
359 		args[2] = Literal.createString(model.getSortMode().name());
360 
361 		FunCall order = new FunCall("Order", args, Syntax.Function);
362 		qa.setExp(order);
363 	}

365 	/**
366 	 * Add Top/BottomCount Funcall to QueryAxis
367 	 * 
368 	 * @param monAx
369 	 * @param nShow
370 	 */
371 	protected void topBottomAxis(ParsedQuery pq, String function) {
372 		// TopCount(TopCount) and TopCount(Order) is not permitted
373 		QueryAxis[] queryAxes = pq.getAxes();
374 		QueryAxis qa = queryAxes[quaxToSort.getOrdinal()];
375 		Exp setForAx = qa.getExp();
376 		Exp sortExp;
377 
378 		List<Member> sortPosMembers = model.getSortPosMembers();
379 		if (sortPosMembers == null) {
380 			return;
381 		}
382 
383 		// if we got more than 1 position member, generate a tuple
384 		if (sortPosMembers.size() > 1) {
385 			Exp[] memberExp = new Exp[sortPosMembers.size()];
386 			for (int i = 0; i < memberExp.length; i++) {
387 				memberExp[i] = QuaxUtil.expForMember(sortPosMembers.get(i));
388 			}
389 			sortExp = new FunCall("()", memberExp, Syntax.Parentheses);
390 		} else {
391 			sortExp = QuaxUtil.expForMember(sortPosMembers.get(0));
392 		}
393 
394 		Exp[] args = new Exp[3];
395 		args[0] = setForAx; // the set representing the query axis
396 		args[1] = Literal.create(model.getTopBottomCount());
397 		args[2] = sortExp;
398 
399 		FunCall topbottom = new FunCall(function, args, Syntax.Function);
400 		qa.setExp(topbottom);
401 	}

403 	/**
404 	 * Swap axes in parsed query
405 	 */
406 	protected void swapAxes() {
407 		QueryAxis[] queryAxes = parsedQuery.getAxes();
408 		if (queryAxes.length >= 2) {
409 			Exp exp = queryAxes[0].getExp();
410 			queryAxes[0].setExp(queryAxes[1].getExp());
411 			queryAxes[1].setExp(exp);
412 		}
413 	}

415 	/**
416 	 * @param mdxQuery
417 	 */
418 	protected ParsedQuery parseQuery(String mdxQuery) {
419 		Reader reader = new StringReader(mdxQuery);
420 		Parser parser = new Parser(new Lexer(reader));
421 
422 		ParsedQuery parsedQuery;
423 
424 		try {
425 			Symbol parseTree = parser.parse();
426 			parsedQuery = (ParsedQuery) parseTree.value;
427 		} catch (RuntimeException e) {
428 			throw e;
429 		} catch (Exception e) {
430 			throw new PivotException(e);
431 		}
432 
433 		parsedQuery.afterParse();
434 
435 		return parsedQuery;
436 	}

438 	/**
439 	 * After the startup query was run: get the current positions as array of
440 	 * array of member. Called from Model.getResult after the query was
441 	 * executed.
442 	 * 
443 	 * @param result
444 	 *            the result which redefines the query axes
445 	 */
446 	public void afterExecute(CellSet cellSet) {
447 		List<CellSetAxis> axes = cellSet.getAxes();
448 
449 		// initialization: get the result positions and set it to quax
450 		// if the quaxes are not yet used to generate the query
451 		if (!useQuax) {
452 			int i = 0;
453 			for (CellSetAxis axis : axes) {
454 				List<Position> positions = axis.getPositions();
455 
456 				int index = axesSwapped ? (i + 1) % 2 : i;
457 				quaxes.get(index).initialize(positions);
458 
459 				i++;
460 			}
461 		} else {
462 			// hierarchize result if neccessary
463 			int i = 0;
464 			for (Quax quax : quaxes) {
465 				int index = axesSwapped ? (i + 1) % 2 : i;
466 				List<Position> positions = axes.get(index).getPositions();
467 
468 				// after a result for CalcSet.GENERATE was gotten
469 				// we have to re-initialize the quax,
470 				// so that we can navigate.
471 				if (quax.getGenerateMode() == CalcSetMode.Generate) {
472 					quax.resetGenerate();
473 					quax.initialize(positions);
474 				} else {
475 					// unknown function members are collected
476 					// - always for a "Sticky generate" unknown function
477 					// - on first result for any other unknown function
478 					int nDimension = quax.getNDimension();
479 					for (int j = 0; j < nDimension; j++) {
480 						// collect members for unknown functions on quax
481 						if (quax.isUnknownFunction(j)) {
482 							List<Member> members = memListForHier(j, positions);
483 							quax.setHierMemberList(j, members);
484 						}
485 					}
486 				}
487 				i++;
488 			}
489 		}
490 
491 		if (logger.isDebugEnabled()) {
492 			// print the result positions to logger
493 			for (CellSetAxis axis : axes) {
494 				List<Position> positions = axis.getPositions();
495 				logger.debug("Positions of axis "
496 						+ axis.getAxisOrdinal().axisOrdinal());
497 
498 				if (positions.size() == 0) {
499 					// the axis does not have any positions
500 					logger.debug("0 positions");
501 				} else {
502 					int nDimension = positions.get(0).getMembers().size();
503 					for (Position position : positions) {
504 						List<Member> members = position.getMembers();
505 
506 						StringBuilder sb = new StringBuilder();
507 						for (int j = 0; j < nDimension; j++) {
508 							if (j > 0) {
509 								sb.append(" * ");
510 							}
511 
512 							List<Member> memsj = new ArrayList<Member>(j + 1);
513 							for (int k = 0; k <= j; k++) {
514 								memsj.add(members.get(k));
515 							}
516 
517 							if (this.canExpand(memsj)) {
518 								sb.append("(+)");
519 							} else if (this.canCollapse(memsj)) {
520 								sb.append("(-)");
521 							} else {
522 								sb.append("   ");
523 							}
524 
525 							sb.append(members.get(j).getUniqueName());
526 						}
527 						logger.debug(sb.toString());
528 					}
529 				}
530 			}
531 		}
532 	}

570 	/**
571 	 * Find out, whether a member can be expanded. this is true, if - the member
572 	 * is on an axis and - the member is not yet expanded and - the member has
573 	 * children
574 	 * 
575 	 * @param member
576 	 *            Member to be expanded
577 	 * @return true if the member can be expanded
578 	 */
579 	public boolean canExpand(Member member) {
580 		// a calculated member cannot be expanded
581 		if (member.isCalculated()) {
582 			return false;
583 		}
584 
585 		try {
586 			if (member.getChildMemberCount() <= 0) {
587 				return false;
588 			}
589 		} catch (OlapException e) {
590 			throw new PivotException(e);
591 		}
592 
593 		Dimension dim = member.getLevel().getHierarchy().getDimension();
594 		Quax quax = findQuax(dim);
595 		return (quax == null) ? false : quax.canExpand(member);
596 	}
