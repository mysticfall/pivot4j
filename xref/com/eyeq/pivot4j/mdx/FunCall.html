72  	/**
73  	 * format to MDX
74  	 */
75  	public String toMdx() {
76  		if ((this.isCallTo("Parameter") || this.isCallTo("ParamRef"))
77  				&& (pQuery.getParaMap().size() > 0)) {
78  			// parameters are evaluated to MDX
79  			return evaluateParameter();
80  		}
81  
82  		StringBuffer sb = new StringBuffer();
83  
84  		// "+" instead of Union yields much better readable MDX
85  		// however
86  		// - does not work with SAP
87  		// - is not compatible with Mondrian
88  		/*
89  		 * sorry if (this.isCallTo("Union")) { // render Union as "+"
90  		 * sb.append(args[0].toMdx()); sb.append(" + ");
91  		 * sb.append(args[1].toMdx()); return sb.toString(); }
92  		 */
93  
94  		boolean isFollow = false;
95  
96  		switch (syntacticType) {
97  		case Function: // f(a, b, c)
98  			sb.append(function);
99  			sb.append("(");
100 			for (int i = 0; i < args.length; i++) {
101 				if (isFollow)
102 					sb.append(", ");
103 				isFollow = true;
104 				sb.append(args[i].toMdx());
105 			}
106 			sb.append(")");
107 			break;
108 		case Braces: // { a, b, c }
109 			sb.append("{");
110 			for (int i = 0; i < args.length; i++) {
111 				if (isFollow)
112 					sb.append(", ");
113 				isFollow = true;
114 				sb.append(args[i].toMdx());
115 			}
116 			sb.append("}");
117 			break;
118 		case Parentheses: // (a, b, c)
119 			sb.append("(");
120 			for (int i = 0; i < args.length; i++) {
121 				if (isFollow)
122 					sb.append(", ");
123 				isFollow = true;
124 				sb.append(args[i].toMdx());
125 			}
126 			sb.append(")");
127 			break;
128 		case Prefix: // NOT a
129 			sb.append(function);
130 			sb.append(" ");
131 			sb.append(args[0].toMdx());
132 			break;
133 		case Infix: // a + b
134 			sb.append(args[0].toMdx());
135 			sb.append(" ");
136 			sb.append(function);
137 			sb.append(" ");
138 			sb.append(args[1].toMdx());
139 			break;
140 		case Property: // a.b
141 		case PropertyQuoted:
142 		case PropertyAmpQuoted:
143 			sb.append(args[0].toMdx());
144 			sb.append(".");
145 			sb.append(function);
146 			break;
147 		case Method:
148 			sb.append(args[0].toMdx());
149 			sb.append(".");
150 			sb.append(function);
151 			sb.append("(");
152 			sb.append(args[1].toMdx());
153 			sb.append(")");
154 			break;
155 		default:
156 			throw new IllegalArgumentException(
157 					"unexpected FunCall syntatic type");
158 		}
159 		return sb.toString();
160 	}

200 	/**
201 	 * put parameter value to MDX
202 	 */
203 	private String evaluateParameter() {
204 		Literal eName = (Literal) args[0];
205 		String paraName = eName.stringValue();
206 		Parameter param = (Parameter) pQuery.paraMap
207 				.get(paraName.toUpperCase());
208 		if (param == null) {
209 			// should not occur
210 			logger.error("could not find parameter " + paraName);
211 			return ("Parameter( \"" + paraName + "\" )"); // MDX parse will fail
212 															// here
213 		}
214 		int type = param.getType();
215 		if (type == Parameter.TYPE_NUMERIC) {
216 			Object value = param.getOValue();
217 			return value.toString(); // Integer or double
218 		} else if (type == Parameter.TYPE_STRING) {
219 			String str = (String) param.getOValue();
220 			return "\"" + str + "\"";
221 		} else {
222 			// member assumed
223 			String str = (String) param.getOValue();
224 			return str;
225 
226 		}
227 	}
