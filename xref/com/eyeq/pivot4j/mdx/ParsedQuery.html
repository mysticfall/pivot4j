141 	public String toMdx() {
142 		StringBuffer mdx = new StringBuffer();
143 		boolean isFollow;
144 		if (formulas.size() > 0) {
145 			mdx.append("WITH ");
146 			for (Formula form : formulas) {
147 				mdx.append(' ');
148 				mdx.append(form.toMdx());
149 			}
150 			mdx.append(' ');
151 		}
152 		mdx.append("SELECT ");
153 		isFollow = false;
154 		for (int i = 0; i < axes.length; i++) {
155 			QueryAxis qa = axes[i];
156 			if (isFollow) {
157 				mdx.append(", ");
158 			}
159 			isFollow = true;
160 			mdx.append(qa.toMdx());
161 		}
162 		mdx.append(" FROM ");
163 		mdx.append(cube);
164 
165 		if (slicer != null) {
166 			mdx.append(" WHERE ");
167 			mdx.append(slicer.toMdx());
168 		}
169 
170 		// add CELL PROPERTIES VALUE, FORMATTED_VALUE, ...
171 		if (cellProps.size() > 0) {
172 			mdx.append(" CELL PROPERTIES VALUE, FORMATTED_VALUE");
173 			for (CompoundId cid : cellProps) {
174 				String str = cid.toMdx();
175 				if (str.equalsIgnoreCase("VALUE"))
176 					continue; // default
177 				if (str.equalsIgnoreCase("FORMATTED_VALUE"))
178 					continue; // default
179 				mdx.append(" ,");
180 				mdx.append(str);
181 			}
182 		}
183 
184 		if (!sapVariables.isEmpty()) {
185 			mdx.append(" SAP VARIABLES ");
186 
187 			boolean first = true;
188 
189 			for (Object sapVariable : sapVariables) {
190 				if (first) {
191 					first = false;
192 				} else {
193 					mdx.append(", ");
194 				}
195 
196 				mdx.append(sapVariable);
197 			}
198 		}
199 
200 		return mdx.toString();
201 	}

310 	/**
311 	 * Collect Parameters and Parameter references
312 	 */
313 	private void collectParams() throws PivotException {
314 		// walk Formulas
315 		int iAxis = -2;
316 		for (Formula formula : formulas) {
317 			walkTreeForParams(formula.getExp(), iAxis);
318 			// dont forget the member properties
319 			for (int i = 0; i < formula.memberProperties.length; i++) {
320 				walkTreeForParams(formula.memberProperties[i].getExp(), iAxis);
321 			}
322 		}
323 
324 		// walk axes
325 		for (int i = 0; i < axes.length; i++) {
326 			Exp exp = axes[i].getExp();
327 			iAxis = i;
328 			walkTreeForParams(exp, iAxis);
329 		}
330 
331 	}

333 	/**
334 	 * Collect Parameters and Parameter references in a subtree below (and
335 	 * including) exp visitor ???
336 	 */
337 	private void walkTreeForParams(Exp exp, int iAxis) throws PivotException {
338 		if (!(exp instanceof FunCall))
339 			return;
340 		FunCall f = (FunCall) exp;
341 		if (f.isCallTo("Parameter")) {
342 			int nArgs = f.getArgs().length;
343 			if (nArgs != 3 && nArgs != 4) {
344 				throw new PivotException(
345 						"Syntax Error in Parameter: wrong number of arguments");
346 			}
347 			if (!(f.getArgs()[0] instanceof Literal)) {
348 				throw new PivotException(
349 						"Syntax Error in Parameter definition - 1.argument");
350 			}
351 
352 			Literal eName = (Literal) f.getArgs()[0];
353 			String paraName = eName.stringValue();
354 			// Found a parameter
355 			// if it is already there, throw an error
356 			if (paraMap.containsKey(paraName.toUpperCase())) {
357 				// already there, error
358 				String err = "Parameter defined more than once:" + paraName;
359 				logger.error(err);
360 				throw new PivotException(err);
361 			} else {
362 				// analyze and add the Parameter
363 				int type;
364 				Object value = null;
365 				f.pQuery = this;
366 				// second Parameter must be ID, either a hierarchy or NUMERIC or
367 				// STRING
368 				CompoundId id = (CompoundId) f.getArgs()[1];
369 				String[] ids = id.toStringArray();
370 				if (ids.length > 1) {
371 					// error, must be NUMERIC or STRING or name of an hierarchy
372 					throw new PivotException(
373 							"Syntax Error in Parameter definition - 2.argument");
374 				}
375 				if (ids[0].equalsIgnoreCase("NUMERIC")) {
376 					type = Parameter.TYPE_NUMERIC;
377 					// 3.argument (value) is literal
378 					if (!(f.getArgs()[0] instanceof Literal)) {
379 						throw new PivotException(
380 								"Syntax Error in Parameter definition - 3.argument");
381 					}
382 					Literal val = (Literal) f.getArgs()[2];
383 					value = val.getValueObject();
384 
385 				} else if (ids[0].equalsIgnoreCase("STRING")) {
386 					type = Parameter.TYPE_STRING;
387 					Literal val = (Literal) f.getArgs()[2];
388 					value = val.getValueObject();
389 				} else {
390 					// hierarchy expected
391 					type = Parameter.TYPE_MEMBER;
392 					CompoundId val = (CompoundId) f.getArgs()[2];
393 					value = val.toMdx(); // member String
394 				}
395 				Parameter par = new Parameter(paraName, type, iAxis);
396 				par.setOValue((Serializable) value);
397 				if (nArgs == 4) {
398 					// set description
399 					Literal desc = (Literal) f.getArgs()[3];
400 					String description = (String) desc.getValueObject();
401 					par.setDescription(description);
402 				}
403 				paraMap.put(paraName.toUpperCase(), par);
404 			}
405 		} else if (f.isCallTo("ParamRef")) {
406 			f.pQuery = this;
407 		}
408 
409 		// recurse down on params
410 		for (int i = 0; i < f.getArgs().length; i++) {
411 			walkTreeForParams(f.getArgs()[i], iAxis);
412 		}
413 
414 	}
