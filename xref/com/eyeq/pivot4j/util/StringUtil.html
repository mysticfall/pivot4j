9   package com.eyeq.pivot4j.util;

11  public class StringUtil {
12  
13  	private StringUtil() {
14  	}

16  	/**
17  	 * Put brackets around string, if not yet there
18  	 * 
19  	 * @param orig
20  	 * @return String
21  	 */
22  	public static String bracketsAround(String orig) {
23  		if (orig.startsWith("[") && orig.endsWith("]")) {
24  			return orig;
25  		} else {
26  			return "[" + orig + "]";
27  		}
28  	}
29  
30  	/**
31  	 * split a unique name
32  	 * 
33  	 * @param uniqueName
34  	 * @return the name parts without brackets
35  	 */
36  	public static String[] splitUniqueName(String uniqueName) {
37  		// uniqueName = [Product].[All Products].[Drink]
38  		String str = uniqueName.trim();
39  		int l2 = str.length() - 1;
40  		if (str.charAt(0) != '[' || str.charAt(l2) != ']')
41  			return new String[] { uniqueName }; // should not occur
42  		// remove first opening bracket and last closing bracket
43  		str = str.substring(1, l2);
44  		// str = Product].[All Products].[Drink
45  		String[] nameParts = str.split("\\]\\.\\[");
46  		return nameParts;
47  	}
48  
49  	/**
50  	 * create unique name from String array
51  	 * 
52  	 * @param strs
53  	 *            - name parts
54  	 * @param n
55  	 *            - number of name parts, all if n &lt;= 0
56  	 * @return unique name
57  	 */
58  	public static String createUName(String[] strs, int n) {
59  		if (n <= 0)
60  			n = strs.length;
61  		StringBuffer sb = new StringBuffer();
62  		for (int i = 0; i < n; i++) {
63  			if (i > 0)
64  				sb.append('.');
65  			sb.append('[');
66  			sb.append(strs[i]);
67  			sb.append(']');
68  		}
69  		return sb.toString();
70  	}
71  
72  	/**
73  	 * extract dimension (first part) from unique name
74  	 * 
75  	 * @param uName
76  	 *            - unique name
77  	 * @return first name part, with brackets
78  	 */
79  	public static String dimFromUName(String uName) {
80  		String[] strs = splitUniqueName(uName);
81  		return "[" + strs[0] + "]";
82  	}
83  
84  	/**
85  	 * extract parent (all exept last part) from unique name
86  	 * 
87  	 * @param uName
88  	 *            - unique name
89  	 * @return all name parts except last, with brackets
90  	 */
91  	public static String parentFromUName(String uName) {
92  		String[] strs = splitUniqueName(uName);
93  		int n = strs.length;
94  		if (n < 3) {
95  			return null; // at least 3 parts required, if a parent exists
96  		}
97  		return createUName(strs, n - 1);
98  	}
