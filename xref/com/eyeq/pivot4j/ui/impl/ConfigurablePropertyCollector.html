9   package com.eyeq.pivot4j.ui.impl;
10  
11  import java.util.ArrayList;
12  import java.util.List;
13  
14  import org.apache.commons.lang.NullArgumentException;
15  import org.olap4j.metadata.Level;
16  import org.olap4j.metadata.NamedList;
17  import org.olap4j.metadata.Property;
18  
19  import com.eyeq.pivot4j.ui.PropertyCollector;
20  
21  public class ConfigurablePropertyCollector implements PropertyCollector {
22  
23  	private List<String> propertyNames;
24  
25  	/**
26  	 * @param propertyNames
27  	 */
28  	public ConfigurablePropertyCollector(List<String> propertyNames) {
29  		if (propertyNames == null) {
30  			throw new NullArgumentException("propertyNames");
31  		}
32  
33  		this.propertyNames = propertyNames;
34  	}
35  
36  	/**
37  	 * @return the propertyNames
38  	 */
39  	protected List<String> getPropertyNames() {
40  		return propertyNames;
41  	}
42  
43  	/**
44  	 * @see com.eyeq.pivot4j.ui.PropertyCollector#getProperties(org.olap4j.metadata.Level)
45  	 */
46  	@Override
47  	public List<Property> getProperties(Level level) {
48  		List<Property> selection = new ArrayList<Property>(propertyNames.size());
49  		NamedList<Property> properties = level.getProperties();
50  
51  		for (String name : propertyNames) {
52  			Property property = properties.get(name);
53  			if (property != null) {
54  				selection.add(property);
55  			}
56  		}
57  
58  		return selection;
59  	}
60  }
