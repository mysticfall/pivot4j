206 	/**
207 	 * @param context
208 	 * @return
209 	 */
210 	protected Double getCellValue(RenderContext context) {
211 		Position position = getPosition(context);
212 
213 		Cell cell = null;
214 
215 		if (position == null) {
216 			cell = context.getCell();
217 		} else if (position.getMembers() != null
218 				&& !position.getMembers().isEmpty()) {
219 			List<Position> positions = new ArrayList<Position>(context
220 					.getAxes().size());
221 
222 			for (Axis axis : context.getAxes()) {
223 				Position pos = context.getPosition(axis);
224 
225 				if (pos == null) {
226 					return null;
227 				}
228 
229 				if (pos.getMembers() != null
230 						&& !pos.getMembers().isEmpty()
231 						&& OlapUtils.equals(pos.getMembers().get(0), position
232 								.getMembers().get(0))) {
233 					pos = position;
234 				}
235 
236 				positions.add(pos);
237 			}
238 
239 			cell = context.getCellSet().getCell(
240 					positions.toArray(new Position[positions.size()]));
241 		}
242 
243 		if (cell == null) {
244 			return null;
245 		}
246 
247 		try {
248 			return cell.getDoubleValue();
249 		} catch (OlapException e) {
250 			throw new PivotException(e);
251 		}
252 	}

254 	/**
255 	 * @param context
256 	 * @return
257 	 */
258 	protected Position getPosition(RenderContext context) {
259 		Position position = null;
260 
261 		if (positionUniqueNames != null && !positionUniqueNames.isEmpty()) {
262 			String cacheKey = "cellValueCache" + hashCode();
263 
264 			if (context.hasAttribute(cacheKey)) {
265 				position = (Position) context.getAttribute(cacheKey);
266 			}
267 
268 			if (position == null) {
269 				for (Axis axis : context.getAxes()) {
270 					Position pos = context.getPosition(axis);
271 
272 					if (pos != null && matches(pos)) {
273 						position = pos;
274 						break;
275 					}
276 				}
277 
278 				if (position != null) {
279 					context.setAttribute(cacheKey, position);
280 				}
281 			}
282 		}
283 
284 		return position;
285 	}

287 	/**
288 	 * @param position
289 	 * @return
290 	 */
291 	protected boolean matches(Position position) {
292 		if (positionUniqueNames == null || position == null) {
293 			return false;
294 		}
295 
296 		boolean matches = false;
297 
298 		List<Member> members = position.getMembers();
299 
300 		if (positionUniqueNames.size() == members.size()) {
301 			Iterator<String> it = positionUniqueNames.iterator();
302 
303 			for (Member member : position.getMembers()) {
304 				matches = it.next().equals(member.getUniqueName());
305 
306 				if (!matches) {
307 					break;
308 				}
309 			}
310 		}
311 
312 		return matches;
313 	}
