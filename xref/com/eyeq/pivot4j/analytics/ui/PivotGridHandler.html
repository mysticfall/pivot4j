274 	public void onCubeChange() {
275 		if (StringUtils.isEmpty(cubeName)) {
276 			if (model.isInitialized()) {
277 				model.destroy();
278 			}
279 		} else {
280 			String mdx;
281 
282 			if (OlapUtils.isEmptySetSupported(model.getMetadata())) {
283 				mdx = String.format(
284 						"select {} on columns, {} on rows from [%s]", cubeName);
285 			} else {
286 				mdx = String.format("select from [%s]", cubeName);
287 			}
288 
289 			model.setMdx(mdx);
290 
291 			if (!model.isInitialized()) {
292 				try {
293 					model.initialize();
294 				} catch (Exception e) {
295 					FacesContext context = FacesContext.getCurrentInstance();
296 					ResourceBundle bundle = context.getApplication()
297 							.getResourceBundle(context, "msg");
298 
299 					String title = bundle.getString("error.unhandled.title");
300 					String message = bundle
301 							.getString("error.unhandled.message") + e;
302 
303 					Logger log = LoggerFactory.getLogger(getClass());
304 					if (log.isErrorEnabled()) {
305 						log.error(title, e);
306 					}
307 
308 					context.addMessage(null, new FacesMessage(
309 							FacesMessage.SEVERITY_ERROR, title, message));
310 
311 				}
312 			}
313 		}

316 	public boolean isValid() {
317 		if (model == null || !model.isInitialized()) {
318 			return false;
319 		}
320 
321 		CellSet cellSet = model.getCellSet();
322 
323 		if (cellSet == null) {
324 			return false;
325 		}
326 
327 		List<CellSetAxis> axes = model.getCellSet().getAxes();
328 		if (axes.size() < 2) {
329 			return false;
330 		}
331 
332 		return axes.get(0).getPositionCount() > 0
333 				&& axes.get(1).getPositionCount() > 0;
334 	}
