109 	public void selectProperty() {
110 		FacesContext context = FacesContext.getCurrentInstance();
111 
112 		Map<String, String> parameters = context.getExternalContext()
113 				.getRequestParameterMap();
114 
115 		PropertyCategory category = PropertyCategory.valueOf(parameters
116 				.get("category"));
117 		String key = parameters.get("key");
118 
119 		setKey(key);
120 		setCategory(category);
121 
122 		this.descriptor = descriptorFactory.getDescriptor(category, key);
123 
124 		RenderPropertyList properties = getProperties(descriptor.getCategory());
125 
126 		SimpleRenderProperty property = (SimpleRenderProperty) properties
127 				.getRenderProperty(descriptor.getKey());
128 
129 		if (property == null) {
130 			this.expression = null;
131 		} else {
132 			this.expression = property.getValue();
133 		}
134 
135 		if (expression != null
136 				&& (expression.contains("${") || expression.contains("<#"))) {
137 			setUseExpression(true);
138 		} else {
139 			setUseExpression(false);
140 		}
141 
142 		editorPanel.getChildren().clear();
143 
144 		PropertyEditor editor = descriptor.getEditor();
145 
146 		if (editor == null) {
147 			this.value = property.getValue();
148 		} else {
149 			Application application = context.getApplication();
150 
151 			ExpressionFactory factory = application.getExpressionFactory();
152 
153 			ValueExpression exp = factory.createValueExpression(
154 					context.getELContext(), "#{propertiesHandler.value}",
155 					Object.class);
156 
157 			MethodExpression listener = factory.createMethodExpression(
158 					context.getELContext(),
159 					"#{propertiesHandler.onPropertyChange}", Void.TYPE,
160 					new Class[0]);
161 
162 			editor.createComponent(descriptor, editorPanel, exp, listener,
163 					"button-bar");
164 
165 			this.value = editor.getValue(descriptor, properties);
166 		}
167 
168 		setDirty(false);
169 	}

175 	public void onEditorModeChange() {
176 		RenderPropertyList properties = getProperties(getCategory());
177 		SimpleRenderProperty property = (SimpleRenderProperty) properties
178 				.getRenderProperty(getKey());
179 
180 		if (getUseExpression()) {
181 			if (expression == null) {
182 				if (property == null) {
183 					this.expression = null;
184 				} else {
185 					this.expression = property.getValue();
186 				}
187 			}
188 		} else {
189 			PropertyEditor editor = descriptor.getEditor();
190 
191 			if (editor == null) {
192 				this.value = null;
193 			} else {
194 				this.value = editor.getValue(descriptor, properties);
195 			}
196 		}
197 	}

233 	public void apply() {
234 		RenderPropertyList properties = getProperties(descriptor.getCategory());
235 
236 		if (getUseExpression()) {
237 			if (expression == null) {
238 				properties.removeRenderProperty(descriptor.getKey());
239 			} else {
240 				SimpleRenderProperty property = new SimpleRenderProperty(
241 						descriptor.getKey(), expression);
242 				properties.setRenderProperty(property);
243 			}
244 		} else {
245 			PropertyEditor editor = descriptor.getEditor();
246 			editor.setValue(descriptor, properties, value);
247 		}
248 
249 		setDirty(false);
250 	}
