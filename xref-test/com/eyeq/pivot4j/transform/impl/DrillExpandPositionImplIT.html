60  	@Test
61  	public void testExpandOnPosition() {
62  		DrillExpandPosition transform = getTransform();
63  
64  		PivotModel model = getPivotModel();
65  
66  		CellSet cellSet = model.getCellSet();
67  		assertNotNull("Unable to execute MDX query : " + getInitialQuery(),
68  				cellSet);
69  
70  		CellSetAxis axis = cellSet.getAxes().get(1);
71  		Position position = axis.getPositions().get(0);
72  
73  		Member allMedia = position.getMembers().get(1);
74  
75  		assertEquals(
76  				"Unexpected member at drill position : "
77  						+ allMedia.getCaption(), "All Media",
78  				allMedia.getCaption());
79  
80  		assertFalse("[All Media] should not be collapsible initially",
81  				transform.canCollapse(position, allMedia));
82  		assertTrue("[All Media] should be expandable initially",
83  				transform.canExpand(position, allMedia));
84  
85  		transform.expand(position, allMedia);
86  
87  		assertEquals("Unexpected MDX after drill down : ",
88  				getTransformedQuery(), model.getCurrentMdx());
89  
90  		assertTrue("[All Media] should be collapsible after drill down",
91  				transform.canCollapse(position, allMedia));
92  		assertFalse("[All Media] should not be expandable after drill down",
93  				transform.canExpand(position, allMedia));
94  
95  		transform.collapse(position, allMedia);
96  
97  		assertEquals("Unexpected MDX after collapse : ", getInitialQuery(),
98  				model.getCurrentMdx());
99  
100 		assertFalse("[All Media] should not be collapsible after collapse",
101 				transform.canCollapse(position, allMedia));
102 		assertTrue("[All Media] should be expandable after collapse",
103 				transform.canExpand(position, allMedia));
104 	}

113 		assertNotNull("Unable to execute MDX query : " + getInitialQuery(),

117 		Position position = axis.getPositions().get(0);
118 
119 		Member allMedia = position.getMembers().get(1);
120 
121 		assertEquals(
122 				"Unexpected member at drill position : "
123 						+ allMedia.getCaption(), "All Media",
124 				allMedia.getCaption());
125 
126 		model.setSorting(true);
127 		model.setTopBottomCount(3);
128 		model.setSortCriteria(SortCriteria.BOTTOMCOUNT);
129 
130 		model.sort(axis, position);
131 
132 		assertFalse("[All Media] should not be collapsible initially",
133 				transform.canCollapse(position, allMedia));
134 		assertTrue("[All Media] should be expandable initially",
135 				transform.canExpand(position, allMedia));
136 
137 		transform.expand(position, allMedia);
138 
139 		assertEquals(
140 				"Unexpected MDX after drill down : ",
141 				"SELECT {[Measures].[Unit Sales], [Measures].[Store Cost], [Measures].[Store Sales]} ON COLUMNS, "
142 						+ "BottomCount(Union(Union(CrossJoin({[Time].[1997]}, {[Promotion Media].[All Media]}), "
143 						+ "CrossJoin({[Time].[1997]}, [Promotion Media].[All Media].Children)), {([Time].[1998], "
144 						+ "[Promotion Media].[All Media])}), 3, ([Time].[1997], [Promotion Media].[All Media])) ON ROWS FROM [Sales]",
145 				model.getCurrentMdx());
146 
147 		assertTrue("[All Media] should be collapsible after drill down",
148 				transform.canCollapse(position, allMedia));
149 		assertFalse("[All Media] should not be expandable after drill down",
150 				transform.canExpand(position, allMedia));
151 
152 		transform.collapse(position, allMedia);
153 
154 		assertEquals(
155 				"Unexpected MDX after collapse : ",
156 				"SELECT {[Measures].[Unit Sales], [Measures].[Store Cost], [Measures].[Store Sales]} "
157 						+ "ON COLUMNS, BottomCount({([Time].[1997], [Promotion Media].[All Media]), "
158 						+ "([Time].[1998], [Promotion Media].[All Media])}, 3, ([Time].[1997], [Promotion Media].[All Media])) "
159 						+ "ON ROWS FROM [Sales]", model.getCurrentMdx());
160 
161 		assertFalse("[All Media] should not be collapsible after collapse",
162 				transform.canCollapse(position, allMedia));
163 		assertTrue("[All Media] should be expandable after collapse",
164 				transform.canExpand(position, allMedia));
