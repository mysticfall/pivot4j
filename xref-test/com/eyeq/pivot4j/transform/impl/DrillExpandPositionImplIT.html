66  	@Test
67  	public void testTransform() {
68  		PivotModel model = getPivotModel();
69  		model.setMdx(getInitialQuery());
70  		model.initialize();
71  
72  		DrillExpandPosition transform = model
73  				.getTransform(DrillExpandPosition.class);
74  
75  		assertNotNull("No suitable transform found for "
76  				+ DrillExpandPosition.class, transform);
77  
78  		CellSet cellSet = model.getCellSet();
79  		assertNotNull("Unable to execute MDX query : " + getInitialQuery(),
80  				cellSet);
81  
82  		CellSetAxis axis = cellSet.getAxes().get(1);
83  		Position position = axis.getPositions().get(0);
84  
85  		Member allMedia = position.getMembers().get(1);
86  
87  		assertEquals(
88  				"Unexpected member at drill position : "
89  						+ allMedia.getCaption(), "All Media",
90  				allMedia.getCaption());
91  
92  		assertFalse("[All Media] should not be collapsible initially",
93  				transform.canCollapse(position, allMedia));
94  		assertTrue("[All Media] should be expandable initially",
95  				transform.canExpand(position, allMedia));
96  
97  		transform.expand(position, allMedia);
98  
99  		assertEquals("Unexpected MDX after drill down : ",
100 				getTransformedQuery(), model.getCurrentMdx());
101 
102 		assertTrue("[All Media] should be collapsible after drill down",
103 				transform.canCollapse(position, allMedia));
104 		assertFalse("[All Media] should not be expandable after drill down",
105 				transform.canExpand(position, allMedia));
106 
107 		transform.collapse(position, allMedia);
108 
109 		assertEquals("Unexpected MDX after collapse : ", getInitialQuery(),
110 				model.getCurrentMdx());
111 
112 		assertFalse("[All Media] should not be collapsible after collapse",
113 				transform.canCollapse(position, allMedia));
114 		assertTrue("[All Media] should be expandable after collapse",
115 				transform.canExpand(position, allMedia));
116 	}
